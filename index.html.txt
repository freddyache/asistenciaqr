    <style>
        body { font-family: sans-serif; text-align: center; padding: 20px; }
        #reader { width: 100%; max-width: 500px; margin: auto; }
        #status { margin-top: 20px; font-weight: bold; color: blue; }

<body>
    <h1>Escaneo de Asistencia</h1>
    <div id="reader"></div>
    <div id="status">Esperando escaneo...</div>

    <script>
        const scriptURL = 'TU_URL_DE_APPS_SCRIPT';

        function onScanSuccess(decodedText) {
            document.getElementById('status').innerText = "Procesando registro...";
            
            // Pausar escáner para evitar duplicados
            html5QrcodeScanner.clear();

            fetch(scriptURL, {
                method: 'POST',
                body: JSON.stringify({ id: decodedText })
            })
            .then(response => response.text())
            .then(result => {
                document.getElementById('status').innerText = result;
                // Reiniciar escáner después de 3 segundos
                setTimeout(() => location.reload(), 3000);
            })
            .catch(error => {
                console.error('Error:', error);
                document.getElementById('status').innerText = "Error en el servidor";
            });
        }

        let html5QrcodeScanner = new Html5QrcodeScanner("reader", { fps: 10, qrbox: 250 });
        html5QrcodeScanner.render(onScanSuccess);
    </script>
</body>
